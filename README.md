    $ npm init -y

    $ npm i path react react-dom babel-loader @babel/core @babel/preset-react @babel/preset-env @babel/plugin-proposal-class-properties style-loader css-loader file-loader --save

    $ npm i -D webpack webpack-cli webpack-dev-server html-webpack-plugin

    -- .gitignore
    -- src/index.js

### Step 6

### Step 7
Add this code to the ***webpack.config.js***

    const HtmlWebPackPlugin = require('html-webpack-plugin');
    const webpack = require('webpack');
    const path = require('path');
    module.exports = {
        context: __dirname,
        entry: './src/index.js',
        output: {
            path: path.resolve(__dirname, 'dist'),
            filename: 'main.js',
            publicPath: '/',
        },
        devServer: {
            historyApiFallback: true
        },
        module: {
            rules: [
                {
                    test: /\.(js|jsx)$/,
                    exclude: /node_modules/,
                    use: ['babel-loader']
                }, // for js or jsx files
                {
                    test: /\.css$/,
                    use: ['style-loader', 'css-loader'],
                }, // for css files
                {
                    test: /\.(png|j?g|svg|gif)?$/,
                    use: 'file-loader'
                }, // for images files
                { 
                    test: /\.woff($|\?)|\.woff2($|\?)|\.ttf($|\?)|\.eot($|\?)|\.svg($|\?)/, 
                    use: ['file-loader'] 
                }  // for fonts
            ],
        },
        plugins: [
            new HtmlWebPackPlugin({
                template: path.resolve(__dirname, 'public/index.html'),
                filename: './index.html'
            }),
            new webpack.HotModuleReplacementPlugin()
        ]
    };

### Step 11
